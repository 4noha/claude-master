    @dataclass
    class LimitEvent:
        session_key: str
        level: Literal["approaching", "interrupt", "reached"]
        usage_percent: int
        reset_time_str: str   # "8:30 pm"
        reset_tz: str         # "UTC+9"
  
    class LimitWatcher:
        APPROACHING_THRESHOLD = 80
        INTERRUPT_THRESHOLD   = 90
  
        def __init__(self):
            self._notified: dict[str, str] = {}  # key → last level notified
  
        def check(self, key: str, status: dict) -> LimitEvent | None:
            pct = status.get("usage_percent")
            if pct is None:
                return None
            level = (
                "reached"    if pct >= 100 else
                "interrupt"  if pct >= self.INTERRUPT_THRESHOLD else
                "approaching" if pct >= self.APPROACHING_THRESHOLD else
                None
            )
            if level and self._notified.get(key) != level:
                self._notified[key] = level
                return LimitEvent(key, level, pct,
                                  status.get("reset_time",""), status.get("reset_tz",""))
            return None
